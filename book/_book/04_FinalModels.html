# Define training control

  method = "repeatedcv",

# Train all models at once using caretList / using bestFit hyperparameters from 03_HyperparameterTuning.qmd
set.seed(42)
modelsList_J1 <- caretList(
  as.formula(paste(response, "~ .")),
  data = dd,
  trControl = trained_control,
  tuneList = list(
    ranger = caretModelSpec(
      method = "ranger",
      tuneGrid = expand.grid(
        mtry = 12,
        splitrule = "variance",
        min.node.size = 5
      ),
      importance = "permutation",
      num.trees = 5000
    ),
    gbm = caretModelSpec(
      method = "gbm",
      tuneGrid = expand.grid(
        n.trees = 300,
        interaction.depth = 10,
        shrinkage = 0.1,
        n.minobsinnode = 10
      ),
      verbose = FALSE
    ),
    xgbTree = caretModelSpec(
      method = "xgbTree",
      tuneGrid = expand.grid(
        nrounds = 1000,
        eta = 0.1,
        max_depth = 5,
        gamma = 0,
        colsample_bytree = 0.6,
        min_child_weight = 1,
        subsample = 1
      )
    )
  )
)
